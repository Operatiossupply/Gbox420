277 bool SyncInProgress=false;
278 time_t getNtpTime(){
279  time_t NTPResponse = 0;
280  if(!SyncInProgress){ //blocking calling the sync again in an interrupt
281  SyncInProgress = true;
282  uint32_t LastRefresh = millis();
283  logToSerials(F("Waiting for NTP time (30sec timeout)..."),false);
284  while(NTPResponse == 0 && millis() - LastRefresh < 30000){
285  NTPResponse = EspCmd.GetTime();
286  delay(500);
287  wdt_reset(); //reset watchdog timeout
288  }
289  SyncInProgress = false;
290  if(NTPResponse == 0) {
291  addToLog(F("NTP time sync failed"));
292  sendEmailAlert(F("NTP%20time%20sync%20failed"));
293  }
294  else logToSerials(F("NTP time synchronized"),true);
295  }
296  return NTPResponse;
297 }
